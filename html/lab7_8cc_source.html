42  ret += frame.header().to_string();
43  switch (frame.header().type) {
44  case EthernetHeader::TYPE_IPv4: {
45  InternetDatagram dgram;
46  if (dgram.parse(frame.payload().concatenate()) == ParseResult::NoError) {
47  ret += " " + dgram.header().summary();
48  if (dgram.header().proto == IPv4Header::PROTO_TCP) {
49  TCPSegment tcp_seg;

51  ParseResult::NoError) {

54  }
